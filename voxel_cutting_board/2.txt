public VoxelShape makeShape(){
	VoxelShape shape = VoxelShapes.empty();
	shape = VoxelShapes.join(shape, VoxelShapes.box(0.0625, 0, 0.625, 0.9375, 0.0625, 0.8125), IBooleanFunction.OR);
	shape = VoxelShapes.join(shape, VoxelShapes.box(0.1875, 0, 0.1875, 0.9375, 0.0625, 0.625), IBooleanFunction.OR);
	shape = VoxelShapes.join(shape, VoxelShapes.box(0.0625, 0, 0.1875, 0.1875, 0.0625, 0.375), IBooleanFunction.OR);
	shape = VoxelShapes.join(shape, VoxelShapes.box(0.0625, 0, 0.375, 0.125, 0.0625, 0.625), IBooleanFunction.OR);

	return shape;
}
