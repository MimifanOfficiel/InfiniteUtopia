public VoxelShape makeShape(){
	VoxelShape shape = VoxelShapes.empty();
	shape = VoxelShapes.join(shape, VoxelShapes.box(0.625, 0, 0.0625, 0.8125, 0.0625, 0.9375), IBooleanFunction.OR);
	shape = VoxelShapes.join(shape, VoxelShapes.box(0.1875, 0, 0.0625, 0.625, 0.0625, 0.8125), IBooleanFunction.OR);
	shape = VoxelShapes.join(shape, VoxelShapes.box(0.1875, 0, 0.8125, 0.375, 0.0625, 0.9375), IBooleanFunction.OR);
	shape = VoxelShapes.join(shape, VoxelShapes.box(0.375, 0, 0.875, 0.625, 0.0625, 0.9375), IBooleanFunction.OR);

	return shape;
}
